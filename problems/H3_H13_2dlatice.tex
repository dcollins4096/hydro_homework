
     A bosonic low-energy excitation of a 2d system of dimension
        $L\times L$ can be described by
        \begin{align}
            \rho \dbd{^2 u}{t^2} + C\nabla^4 u=0.
        \end{align}
        \begin{enumerate}
            \item \pts{4}Find the dispersion relation
            \item \pts{2}Find the phase speed
            \item \pts{2}Find the group speed
        \end{enumerate}

\solution{
        \emph{Solution}

        \begin{align}
            u=u_0 e^{-i(k x-\omega t)}\nn\\
           -\rho \omega^2+C k^4 &= 0\nn\\
           \omega &= \sqrt{\frac{C}{\rho}} k^2\\
           v_p=\frac{\omega}{k}&= \sqrt{\frac{C}{rho}} k\nn\\
           v_g=\frac{d \omega}{d k}&=2 \sqrt{\frac{C}{rho}} k\nn\\
        \end{align}

}
