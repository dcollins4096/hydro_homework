
   \pts{8} Derive the Rankine Hugonio equations (6.33, 6.34, and 6.35) from the conservative form of the
    equations (Continuity, Momentum (4.23) and internal energy (4.25)).

\solution{
     \emph{Solution}
 
     Steady state, integrate, algebrate.
 
     \begin{align}
         \dbd{f}{t} + \dbd{g}{x} &= 0 \rm{\ (all\ things)}\nn\\
         0 +\int_a^b \dbd{g}{x} dx &= 0\rm{\ (steady.\ integrate.)}\nn\\
         g(a)&=g(b)\nn
     \end{align}
