
\pts{6} Starting from the continuity equation, find an expression for 
\begin{align}
\ddbd{\rho}{t}
\end{align}

\solution{\emph{Solution}

\begin{align}
	\dbd{\rho}{t} + \dbd{ }{x_j} \rho v_j &= 0\\
	\dbd{\rho}{t} + v_j \dbd{ }{x_j} \rho + \rho \dbd{ }{x_j} v_j &=0\\
	\ddbd{\rho}{t} + \rho \dbd{ }{x_j} v_j &= 0\\
	\ddbd{\rho}{t} &= - \rho \nabla \cdot \vec{v}
\end{align}

}
