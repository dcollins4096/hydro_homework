
     \pts{2} Starting with $c_s^2=\dbd{p}{\rho}$, show that the sound speed, $c_s^2=\gamma\frac{k_B}{m}T$, where $k_B$
      is the Boltzmann constant, $m$ is the mass of whatever atom.  Assume the
      process is adiabatic.

\solution{
       \emph{Solution}
 
       From further in the homework set, 
       \begin{align}
           c_s^2 = \ddbd{P}{\rho}\nn
       \end{align}
       and for an adiabatic gas,
       \begin{align}
           P=k \rho^{\gamma}\nn
       \end{align}
       so
       $$
       c_s^2 = \gamma \rho^{\gamma-1} = \gamma \frac{P}{\rho}\nn
       $$
       Then for an ideal gas, 
       \begin{align}
           P&=\rho \frac{k_{\rm{B}}}{m} T\nn\\
           c_s^2 &=\gamma \frac{k_{\rm{B}}}{m} T\nn\\
       \end{align}
}
